Q2) A) #Write a Python program to create a Pie plot to get the frequency
 #of the three species of the Iris data (Use 1ris.csv)
 # (slip1 and slip11 are same)
 import pandas as pd
 import matplotlib.pyplot as plt
 d = pd.read_csv('Iris.csv')
 ax = plt.figure(figsize=(5, 5))
 d['Species'].value_counts().plot.pie()
 plt.title('Frequency of Iris Species')
 plt.axis('equal')
 plt.show()
