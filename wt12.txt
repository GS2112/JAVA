<?php

$num1 = $_POST['num1']; 
$num2 = $_POST['num2']; 
$operation = $_POST['operation']; 
switch($operation) 
{ 
case "add": 
$add=sum($num1,$num2); 
echo "$add"; 
break; 
case "subtract": 
$sub=sub($num1,$num2); 
echo "$sub"; 
break; 
case "multiply": 
$mul=mul($num1,$num2); 
echo "$mul"; 
break; 
case "divide": 
$div=div($num1,$num2); 
echo "$div";  
break;     
default: 
return "Invalid operation selected."; 
}

//  'myfunctions2.php ‘ 
<?php 
function sum($num1=10,$num2=10) 
{ 
$add=$num1+$num2; 
return $add; 
} 
function sub($num1=10,$num2=20) 
{ 
$sub=$num1-$num2; 
return $sub; 
} 
function mul($num1=10,$num2=20) 
{ 
$mul=$num1*$num2; 
return $mul; 
} 
function div($num1=10,$num2=20) 
{ 
$div=$num1/$num2; 
return $div; 
} 
?>
