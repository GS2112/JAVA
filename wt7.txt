Q2)[3]: # Write a Python program to perform the following tasks :
# a. Apply OneHot coding on Country column.
# b. Apply Label encoding on purchased column
# (Data.csv have two categorical column the country column, and the purchased␣
↪column).
import pandas as pd
from sklearn.preprocessing import LabelEncoder
d = pd.read_csv('s7.csv')
# Apply Label encoding on the 'Purchased' column
le = LabelEncoder()
d['Purchased'] = le.fit_transform(d['Purchased'])
# Apply OneHot encoding on the 'Country' column
ohed = pd.get_dummies(d, columns=['Country'])
print(ohed)
